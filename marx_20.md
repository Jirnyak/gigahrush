#### 2. Слой Systems (Логика) - `src/systems/interactions.ts`
Логика дверей уже поддерживает `keyId`. Убедитесь, что лифтовые двери корректно обрабатывают наличие ключа в инвентаре игрока. При попытке открытия необходимо проверить инвентарь и разблокировать дверь.
```typescript
// Stub for interactions.ts logic check
function handleDoorInteraction(entity: Entity, door: Door) {
  if (door.state === DoorState.LOCKED && door.keyId) {
    const hasKey = inventoryHasItem(entity, door.keyId);
    if (hasKey) {
      door.state = DoorState.CLOSED; // Unlocked successfully
      publishEvent('door_unlocked', { doorId: door.id, keyId: door.keyId });
      // Добавить звуковой эффект разблокировки электронного замка
      publishEvent('noise', { type: 'electronic_unlock', pos: entity.pos });
    } else {
      // Show UI message if key is missing
      publishEvent('ui_message', { text: \`Требуется ключ: \${door.keyId}\` });
    }
  }
}
```
