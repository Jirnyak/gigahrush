```typescript
// src/data/perks.ts
export enum PerkId {
    STRONG_BACK = 'STRONG_BACK',
    CANNIBAL = 'CANNIBAL',
    NIGHT_VISION = 'NIGHT_VISION',
    SILVER_TONGUE = 'SILVER_TONGUE',
    TECHNICIAN = 'TECHNICIAN'
}

export interface PerkDef {
    id: PerkId;
    name: string;
    description: string;
    maxLevel: number;
    requirements?: { stat: string; value: number }[];
    // Мутаторы статов, применяемые при наличии перка
    statModifiers?: Record<string, number>; 
}

export const PERK_REGISTRY: Record<PerkId, PerkDef> = {
    [PerkId.STRONG_BACK]: {
        id: PerkId.STRONG_BACK,
        name: "Крепкая спина",
        description: "+20 к максимальному весу",
        maxLevel: 3,
        statModifiers: { maxWeight: 20 }
    },
    // Добавить остальные...
};
```

### 2. Интеграция в стейт игрока и сущностей

### 3. Логика применения перков (rpg.ts)
- Напишите функцию `hasPerk(entity, perkId, minLevel = 1)` для быстрых проверок в других системах.
- При пересчете статов (например, `recalculateStats`), суммируйте бонусы от `statModifiers` всех активных перков.

```typescript
// src/systems/rpg.ts
export function hasPerk(entity: Entity, perkId: PerkId, minLevel: number = 1): boolean {
    return (entity.perks?.[perkId] ?? 0) >= minLevel;
}
```
